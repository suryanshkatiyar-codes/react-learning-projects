useState:-
let's say u want to change the value of a variable u can change it easily but when u change it directly on the website it will not change u have to use React to change the values u can not directly interact with the DOM

for example-

import React from 'react'

const App = () => {

  let a = 10;

  function increment() {
    console.log(a);
    a++;
    console.log(a);
  }

  return (
    <div>
      <p>The value of a is {a}</p>
      <button onClick={increment}>click</button>
    </div>
  )
}

export default App

to solve this problem use useState in simple useState (in order to use useState u have to import it first) tells react to do the changes for u
for example-

const [num,setNum]=useState // automatically imports it
here num is read only and setNum is write only

import React, { useState } from 'react';

const App = () => {

  const [num,setNum]=useState(10); // can be anything string array etc

  function changeNum() {
    setNum(20);
  }

  return (
    <div>
      <p>The value of num is {num}</p>
      <button onClick={changeNum}>click</button>
    </div>
  )
}

export default App
