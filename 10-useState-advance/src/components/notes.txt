import React, { useState } from 'react'

const App = () => {

  const [num,setNum]=useState({name:"ryu" ,age:20});

  const btnClicked =()=>{
    const newNum={...num};
    newNum.name="l";
    newNum.age=29;
    setNum(newNum);
  }

  return (
    <div>
      <h1>{num.name}{num.age}</h1>
      <button onClick={btnClicked}>click</button>
    </div>
  )
}

export default App
