Asynchronous JavaScript allows code to run without blocking. Time-consuming operations (like fetching data) run in the background, and the rest of the code continues executing.

import React from 'react'

const App = () => {

  const getData=()=>{
    const response=fetch('https://jsonplaceholder.typicode.com/todos/1');
    console.log(response);
  }

  return (
    <div>
      <button onClick={getData}>Click here</button>
    </div>
  )
}

export default App

// snippet 2 here the code will behave synchronously because we used async and await here

import React from 'react'

const App = () => {

  const getData=async()=>{
    const response=await fetch('https://jsonplaceholder.typicode.com/todos/1'); asynchronous
    console.log(response);
    // console.log(response.json()); asynchronous 
    const data=await response.json();
    console.log(data);
  }

  return (
    <div>
      <button onClick={getData}>Click here</button>
    </div>
  )
}

export default App

// snippet 3 api calling using axios

import axios from 'axios'

const App = () => {

  const getData=async()=>{
    const {data}=await axios.get('https://jsonplaceholder.typicode.com/todos/1');
    console.log(data);
  }

  return (
    <div>
      <button onClick={getData}>Click here</button>
    </div>
  )
}

export default App
